161 /**
162 * @brief Lock a spinlock
163 *
164 * This routine locks the specified spinlock, returning a key handle
165 * representing interrupt state needed at unlock time. Upon
166 * returning, the calling thread is guaranteed not to be suspended or
167 * interrupted on its current CPU until it calls k_spin_unlock(). The
168 * implementation guarantees mutual exclusion: exactly one thread on
169 * one CPU will return from k_spin_lock() at a time. Other CPUs
170 * trying to acquire a lock already held by another CPU will enter an
171 * implementation-defined busy loop ("spinning") until the lock is
172 * released.
173 *
174 * Separate spin locks may be nested. It is legal to lock an
175 * (unlocked) spin lock while holding a different lock. Spin locks
176 * are not recursive, however: an attempt to acquire a spin lock that
177 * the CPU already holds will deadlock.
178 *
179 * In circumstances where only one CPU exists, the behavior of
180 * k_spin_lock() remains as specified above, though obviously no
181 * spinning will take place. Implementations may be free to optimize
182 * in uniprocessor contexts such that the locking reduces to an
183 * interrupt mask operation.
184 *
185 * @warning
186 * Holding a spinlock when a context switch occurs is illegal.
187 *
188 * @param l A pointer to the spinlock to lock
189 * @return A key value that must be passed to k_spin_unlock() when the
190 * lock is released.
191 */
192 45 static ALWAYS_INLINE k_spinlock_key_t k_spin_lock(struct k_spinlock *l)
193 {
194 ARG_UNUSED(l);
195 k_spinlock_key_t k;
196
197 /* Note that we need to use the underlying arch-specific lock
198 * implementation. The "irq_lock()" API in SMP context is
199 * actually a wrapper for a global spinlock!
200 */
201 45 k.key = arch_irq_lock();
202
203 45 z_spinlock_validate_pre(l);
204 #ifdef CONFIG_SMP
205 #ifdef CONFIG_TICKET_SPINLOCKS
206 /*
207 * Enqueue ourselves to the end of a spinlock waiters queue
208 * receiving a ticket
209 */
210 atomic_val_t ticket = atomic_inc(&l->tail);
211 /* Spin until our ticket is served */
212 while (atomic_get(&l->owner) != ticket) {
213 arch_spin_relax();
214 }
215 #else
216 while (!atomic_cas(&l->locked, 0, 1)) {
217 do {
218 arch_spin_relax();
219 } while (atomic_get(&l->locked) != 0);
220 }
221 #endif /* CONFIG_TICKET_SPINLOCKS */
222 #endif /* CONFIG_SMP */
223 45 z_spinlock_validate_post(l);
224
225 45 return k;

228 /**
229 * @brief Attempt to lock a spinlock
230 *
231 * This routine makes one attempt to lock @p l. If it is successful, then
232 * it will store the key into @p k.
233 *
234 * @param[in] l A pointer to the spinlock to lock
235 * @param[out] k A pointer to the spinlock key
236 * @retval 0 on success
237 * @retval -EBUSY if another thread holds the lock
238 *
239 * @see k_spin_lock
240 * @see k_spin_unlock
241 */
242 static ALWAYS_INLINE int k_spin_trylock(struct k_spinlock *l, k_spinlock_key_t *k)
243 {
244 int key = arch_irq_lock();
245
246 z_spinlock_validate_pre(l);
247 #ifdef CONFIG_SMP
248 #ifdef CONFIG_TICKET_SPINLOCKS
249 /*
250 * atomic_get and atomic_cas operations below are not executed
251 * simultaneously.
252 * So in theory k_spin_trylock can lock an already locked spinlock.
253 * To reproduce this the following conditions should be met after we
254 * executed atomic_get and before we executed atomic_cas:
255 *
256 * - spinlock needs to be taken 0xffff_..._ffff + 1 times
257 * (which requires 0xffff_..._ffff number of CPUs, as k_spin_lock call
258 * is blocking) or
259 * - spinlock needs to be taken and released 0xffff_..._ffff times and
260 * then taken again
261 *
262 * In real-life systems this is considered non-reproducible given that
263 * required actions need to be done during this tiny window of several
264 * CPU instructions (which execute with interrupt locked,
265 * so no preemption can happen here)
266 */
267 atomic_val_t ticket_val = atomic_get(&l->owner);
268
269 if (!atomic_cas(&l->tail, ticket_val, ticket_val + 1)) {
270 goto busy;
271 }
272 #else
273 if (!atomic_cas(&l->locked, 0, 1)) {
274 goto busy;
275 }
276 #endif /* CONFIG_TICKET_SPINLOCKS */
277 #endif /* CONFIG_SMP */
278 z_spinlock_validate_post(l);
279
280 k->key = key;
281
282 return 0;
283
284 #ifdef CONFIG_SMP
285 busy:
286 arch_irq_unlock(key);
287 return -EBUSY;
288 #endif /* CONFIG_SMP */
289 }

291 /**
292 * @brief Unlock a spin lock
293 *
294 * This releases a lock acquired by k_spin_lock(). After this
295 * function is called, any CPU will be able to acquire the lock. If
296 * other CPUs are currently spinning inside k_spin_lock() waiting for
297 * this lock, exactly one of them will return synchronously with the
298 * lock held.
