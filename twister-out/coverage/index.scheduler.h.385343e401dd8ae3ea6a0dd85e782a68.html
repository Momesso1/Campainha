58 /**
59 * @brief Walks the wait queue invoking a callback on each waiting thread
60 *
61 * This function walks the wait queue invoking the `walk_func` callback function
62 * on each waiting thread (except if stopped earlier by a non-zero return value),
63 * followed by a single call of `post_func`, all while holding `_sched_spinlock`.
64 * This can be useful for routines that need to operate on multiple waiting threads.
65 *
66 * CAUTION! As a wait queue is of indeterminate length, the scheduler will be
67 * locked for an indeterminate amount of time. This may impact system performance.
68 * As such, care must be taken when using this function and in the callbacks.
69 *
70 * @warning @p walk_func may safely remove the thread received as argument from
71 * the wait queue only when `CONFIG_WAITQ_SCALABLE=n`.
72 *
73 * @param wait_q Identifies the wait queue to walk
74 * @param walk_func Callback to invoke for each waiting thread
75 * @param post_func Callback to invoke after queue walk (optional)
76 * @param data Custom data passed to the callbacks
77 *
78 * @return non-zero if walk is terminated by the callback; otherwise 0
79 */
80 int z_sched_waitq_walk(_wait_q_t *wait_q, _waitq_walk_cb_t walk_func,
81 _waitq_post_walk_cb_t post_func, void *data);
