37 /* Spin, with the scheduler lock held (!), on a thread that is known
38 * (!!) to have released the lock and be on a path where it will
39 * deterministically (!!!) reach arch_switch() in very small constant
40 * time.
41 *
42 * This exists to treat an unavoidable SMP race when threads swap --
43 * their thread record is in the queue (and visible to other CPUs)
44 * before arch_switch() finishes saving state. We must spin for the
45 * switch handle before entering a new thread. See docs on
46 * arch_switch().
47 *
48 * Stated differently: there's a chicken and egg bug with the question
49 * of "is a thread running or not?". The thread needs to mark itself
50 * "not running" from its own context, but at that moment it obviously
51 * is still running until it reaches arch_switch()! Locking can't
52 * treat this because the scheduler lock can't be released by the
53 * switched-to thread, which is going to (obviously) be running its
54 * own code and doesn't know it was switched out.
55 */
56 static inline void z_sched_switch_spin(struct k_thread *thread)
57 {
58 #ifdef CONFIG_SMP
59 volatile void **shp = (void *)&thread->switch_handle;
60
61 while (*shp == NULL) {
62 arch_spin_relax();
63 }
64 /* Read barrier: don't allow any subsequent loads in the
65 * calling code to reorder before we saw switch_handle go
66 * non-null.
67 */
68 barrier_dmem_fence_full();
69 #endif /* CONFIG_SMP */
70 }
71
72 /* New style context switching. arch_switch() is a lower level
73 * primitive that doesn't know about the scheduler or return value.
74 * Needed for SMP, where the scheduler requires spinlocking that we
75 * don't want to have to do in per-architecture assembly.
76 *
77 * Note that is_spinlock is a compile-time construct which will be
78 * optimized out when this function is expanded.
79 */
80 static ALWAYS_INLINE unsigned int do_swap(unsigned int key,
81 struct k_spinlock *lock,
82 bool is_spinlock)
83 {
84 struct k_thread *new_thread, *old_thread;
85
86 #ifdef CONFIG_SPIN_VALIDATE
87 z_assert_can_swap(key, lock);
88 #endif /* CONFIG_SPIN_VALIDATE */
89
90 old_thread = _current;
91
92 z_check_stack_sentinel();
93
94 old_thread->swap_retval = -EAGAIN;
95
96 /* We always take the scheduler spinlock if we don't already
97 * have it. We "release" other spinlocks here. But we never
98 * drop the interrupt lock.
99 */
100 if (is_spinlock && lock != NULL && lock != &_sched_spinlock) {
101 k_spin_release(lock);
102 }
103 if (IS_ENABLED(CONFIG_SMP) || IS_ENABLED(CONFIG_SPIN_VALIDATE)) {
104 /* Taking a nested uniprocessor lock in void context is a noop */
105 if (!is_spinlock || lock != &_sched_spinlock) {
106 (void)k_spin_lock(&_sched_spinlock);
107 }
108 }
109
110 #ifdef CONFIG_SMP
111 new_thread = z_swap_next_thread();
112 #else
113 new_thread = _kernel.ready_q.cache;
114 #endif
115
116 if (new_thread != old_thread) {
117 z_sched_usage_switch(new_thread);
118
119 #ifdef CONFIG_SMP
120 new_thread->base.cpu = arch_curr_cpu()->id;
121
122 if (!is_spinlock) {
123 z_smp_release_global_lock(new_thread);
124 }
125 #endif /* CONFIG_SMP */
126 z_thread_mark_switched_out();
127 z_sched_switch_spin(new_thread);
128 z_current_thread_set(new_thread);
129
130 #ifdef CONFIG_TIMESLICING
131 z_time_slice_reset(new_thread);
132 #endif /* CONFIG_TIMESLICING */
133
134 #ifdef CONFIG_SPIN_VALIDATE
135 z_spin_lock_transfer_owner(&_sched_spinlock);
136 #endif /* CONFIG_SPIN_VALIDATE */
137
138 arch_cohere_stacks(old_thread, NULL, new_thread);
139
140 #ifdef CONFIG_SMP
141 /* Now add _current back to the run queue, once we are
142 * guaranteed to reach the context switch in finite time. See
143 * z_sched_switch_spin(). The current thread can never live in
144 * the run queue until we are inexorably on the context switch
145 * path on SMP, otherwise there is a deadlock condition where a
146 * set of CPUs pick a cycle of threads to run and wait for them
147 * all to context switch forever.
148 */
149 if (z_is_thread_queued(old_thread)) {
150 runq_add(old_thread);
151 }
152 signal_pending_ipi();
153 #endif /* CONFIG_SMP */
154 void *newsh = new_thread->switch_handle;
155
156 if (IS_ENABLED(CONFIG_SMP)) {
157 /* Active threads must have a null here. And
158 * it must be seen before the scheduler lock
159 * is released!
160 */
161 new_thread->switch_handle = NULL;
162 barrier_dmem_fence_full(); /* write barrier */
163 }
164 k_spin_release(&_sched_spinlock);
165 arch_switch(newsh, &old_thread->switch_handle);
166 } else {
167 k_spin_release(&_sched_spinlock);
168 }
169
170 if (is_spinlock) {
171 arch_irq_unlock(key);
172 } else {
173 irq_unlock(key);
174 }
175
176 return _current->swap_retval;
177 }
