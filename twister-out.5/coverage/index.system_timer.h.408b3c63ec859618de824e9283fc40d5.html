26 /*
27 * Maximum number of ticks the kernel will ever ask a timer driver to wait
28 * before the next sys_clock_announce(). It is half of the unsigned tick
29 * range so that the elapsed-tick count the driver eventually announces is
30 * guaranteed to fit in the (unsigned) sys_clock_announce() argument. The
31 * other half is left as slack to absorb a late announce (e.g. interrupt
32 * latency, or a timeout that fires slightly past its deadline) without
33 * overflowing that argument. The kernel caps the value passed to
34 * sys_clock_set_timeout() to this, so a driver need not clamp against the
35 * announce range and only has to honour its own cycle-count limits.
36 */
37 #define SYS_CLOCK_MAX_WAIT (UINT32_MAX / 2)

45 /**
46 * @brief Lock the system clock.
47 *
48 * Acquires the kernel timer lock that protects tick accounting and
49 * the timeout queue. Timer drivers should call this at the start of
50 * their ISR and pass the returned key to sys_clock_announce_locked()
51 * which consumes it. The lock is released when
52 * sys_clock_announce_locked() returns.
53 *
54 * The driver-provided functions sys_clock_set_timeout() and
55 * sys_clock_elapsed() are always called by the kernel with this lock
56 * already held.
57 *
58 * Example usage from a timer ISR:
59 *
60 * @code{.c}
61 * static void timer_isr(const void *arg)
62 * {
63 * k_spinlock_key_t key = sys_clock_lock();
64 *
65 * // Update driver-private state (e.g. cycle counter baseline)
66 * uint64_t now = read_hw_counter();
67 * uint32_t dticks = (now - last_cycle) / CYC_PER_TICK;
68 * last_cycle += (uint64_t)dticks * CYC_PER_TICK;
69 *
70 * // Reprogram comparator if needed ...
71 *
72 * // Announce ticks — key ownership transfers to announce.
73 * sys_clock_announce_locked(dticks, key);
74 * }
75 * @endcode
76 *
77 * @return Lock key to be passed to sys_clock_announce_locked()
78 * or sys_clock_unlock().
79 */
80 #if defined(CONFIG_SMP) || defined(CONFIG_SPIN_VALIDATE)
81 k_spinlock_key_t sys_clock_lock(void);
82 #else
83 /*
84 * When actual spinlocks are not needed (UP without CONFIG_SPIN_VALIDATE),
85 * k_spin_lock() reduces to arch_irq_lock() and the lock argument is
86 * ignored. Inline this to avoid the overhead of an extra function
87 * call for legacy drivers using sys_clock_announce().
88 */
89 static inline k_spinlock_key_t sys_clock_lock(void)
90 {
91 k_spinlock_key_t key;
92
93 /* If this fires, a new config grew real spinlock content and
94 * the #if guard above needs updating.
95 */
96 BUILD_ASSERT(sizeof(struct k_spinlock) <= 1);
97
98 key.key = arch_irq_lock();
99 return key;
100 }
101 #endif

136 /**
137 * @brief Set system clock timeout
138 *
139 * Informs the system clock driver that the next needed call to
140 * sys_clock_announce() will not be until the specified number of ticks
141 * from the current time have elapsed. Note that spurious calls
142 * to sys_clock_announce() are allowed (i.e. it's legal to announce
143 * every tick and implement this function as a noop), the requirement
144 * is that one tick announcement should occur within one tick BEFORE
145 * the specified expiration (that is, passing ticks==1 means "announce
146 * the next tick", this convention was chosen to match legacy usage).
147 * Similarly a ticks value of zero is legal: it simply indicates the
148 * kernel would like the next tick announcement as soon as possible.
149 *
150 * Note that ticks can also be passed the special value K_TICKS_FOREVER,
151 * indicating that no future timer interrupts are expected or required
152 * and that the system is permitted to enter an indefinite sleep even
153 * if this could cause rollover of the internal counter (i.e. the
154 * system uptime counter is allowed to be wrong
155 *
156 * Note also that it is conventional for the kernel to pass INT_MAX
157 * for ticks if it wants to preserve the uptime tick count but doesn't
158 * have a specific event to await. The intent here is that the driver
159 * will schedule any needed timeout as far into the future as
160 * possible. For the specific case of INT_MAX, the next call to
161 * sys_clock_announce() may occur at any point in the future, not just
162 * at INT_MAX ticks. But the correspondence between the announced
163 * ticks and real-world time must be correct.
164 *
165 * A final note about SMP: note that the call to sys_clock_set_timeout()
166 * is made on any CPU, and reflects the next timeout desired globally.
167 * The resulting calls(s) to sys_clock_announce() must be properly
168 * serialized by the driver such that a given tick is announced
169 * exactly once across the system. The kernel does not (cannot,
170 * really) attempt to serialize things by "assigning" timeouts to
171 * specific CPUs.
172 *
173 * @note This function is called by the kernel with the system clock
174 * lock held.
175 *
176 * @param ticks Timeout in tick units
177 * @param idle Hint to the driver that the system is about to enter
178 * the idle state immediately after setting the timeout
179 */
180 void sys_clock_set_timeout(uint32_t ticks, bool idle);

197 * @brief Announce time progress to the kernel

199 * Informs the kernel that the specified number of ticks have elapsed

255 /**
256 * @brief Hardware cycle counter
257 *
258 * Timer drivers are generally responsible for the system cycle
259 * counter as well as the tick announcements. This function is
260 * generally called out of the architecture layer (@see
261 * arch_k_cycle_get_32()) to implement the cycle counter, though the
262 * user-facing API is owned by the architecture, not the driver. The
263 * rate must match CONFIG_SYS_CLOCK_HW_CYCLES_PER_SEC.
264 *
265 * @note
266 * If the counter clock is large enough for this to wrap its full range
267 * within a few seconds (i.e. CONFIG_SYS_CLOCK_HW_CYCLES_PER_SEC is greater
268 * than 50Mhz) then it is recommended to also implement
269 * sys_clock_cycle_get_64().
270 *
271 * @return The current cycle time. This should count up monotonically
272 * through the full 32 bit space, wrapping at 0xffffffff. Hardware
273 * with fewer bits of precision in the timer is expected to synthesize
274 * a 32 bit count.
275 */
276 uint32_t sys_clock_cycle_get_32(void);
