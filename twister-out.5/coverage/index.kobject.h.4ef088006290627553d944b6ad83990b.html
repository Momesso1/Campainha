54 /**
55 * @brief Grant a static thread access to a list of kernel objects
56 *
57 * For threads declared with K_THREAD_DEFINE(), grant the thread access to
58 * a set of kernel objects. These objects do not need to be in an initialized
59 * state. The permissions will be granted when the threads are initialized
60 * in the early boot sequence.
61 *
62 * All arguments beyond the first must be pointers to kernel objects.
63 *
64 * @param name_ Name of the thread, as passed to K_THREAD_DEFINE()
65 */
66 #define K_THREAD_ACCESS_GRANT(name_, ...) \
67 static void * const _CONCAT(_object_list_, name_)[] = \
68 { __VA_ARGS__, NULL }; \
69 static const STRUCT_SECTION_ITERABLE(k_object_assignment, \
70 _CONCAT(_object_access_, name_)) = \
71 { (&_k_thread_obj_ ## name_), \
72 (_CONCAT(_object_list_, name_)) }

83 /**
84 * Grant a thread access to a kernel object
85 *
86 * The thread will be granted access to the object if the caller is from
87 * supervisor mode, or the caller is from user mode AND has permissions
88 * on both the object and the thread whose access is being granted.
89 *
90 * @param object Address of kernel object
91 * @param thread Thread to grant access to the object
92 */
93 __syscall void k_object_access_grant(const void *object,
94 struct k_thread *thread);
95
96 /**
97 * Revoke a thread's access to a kernel object
98 *
99 * The thread will lose access to the object if the caller is from
100 * supervisor mode, or the caller is from user mode AND has permissions
101 * on both the object and the thread whose access is being revoked.
102 *
103 * @param object Address of kernel object
104 * @param thread Thread to remove access to the object
105 */
106 void k_object_access_revoke(const void *object, struct k_thread *thread);
107
108 /**
109 * Revoke access to a kernel object for all other threads
110 *
111 * User threads may grant other threads access to objects they have access to,
112 * including threads they themselves created. In many cases, therefore,
113 * revoking access only from a specified thread is not sufficient. This API
114 * revokes access from all other threads except the current one, so that no
115 * user threads with permissions remain.
116 *
117 * This will also revert public access status, which may have been given to an
118 * object by k_object_access_all_grant() before.
119 *
120 * Note that this function can only be used by kernel threads. Retaining access
121 * by the caller thread therefore makes little difference permission-wise, but
122 * we want to keep one owner so that it is clear that this function never
123 * releases a dynamic object, which would normally happen once it no longer has
124 * any owner.
125 *
126 * @param object Address of kernel object
127 */
128 void k_object_access_revoke_others(const void *object);
129
130 /**
131 * @brief Release an object
132 *
133 * Allows user threads to drop their own permission on an object
134 * Their permissions are automatically cleared when a thread terminates.
135 *
136 * @param object The object to be released
137 *
138 */
139 __syscall void k_object_release(const void *object);
140
141 /**
142 * Grant all present and future threads access to an object
143 *
144 * If the caller is from supervisor mode, or the caller is from user mode and
145 * have sufficient permissions on the object, then that object will have
146 * permissions granted to it for *all* current and future threads running in
147 * the system, effectively becoming a public kernel object.
148 *
149 * Use of this API should be avoided on systems that are running untrusted code
150 * as it is possible for such code to derive the addresses of kernel objects
151 * and perform unwanted operations on them.
152 *
153 * It is not possible to revoke permissions for individual threads on public
154 * objects; once public, any thread may use it. k_object_access_revoke_others()
155 * can be used to revert the public access status of an object.
156 *
157 * @param object Address of kernel object
158 */
159 void k_object_access_all_grant(const void *object);
160
