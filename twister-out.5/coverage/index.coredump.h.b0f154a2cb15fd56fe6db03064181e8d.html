73 /** Command ID */
74 enum coredump_cmd_id {
75 /**
76 * Clear error code from backend.
77 *
78 * Returns 0 if successful, failed otherwise.
79 */
80 COREDUMP_CMD_CLEAR_ERROR,
81
82 /**
83 * Verify that the stored coredump is valid.
84 *
85 * Returns:
86 * - 1 if valid.
87 * - 0 if not valid or no stored coredump.
88 * - -ENOTSUP if this command is not supported.
89 * - Otherwise, error code from backend.
90 */
91 COREDUMP_CMD_VERIFY_STORED_DUMP,
92
93 /**
94 * Erase the stored coredump.
95 *
96 * Returns:
97 * - 0 if successful.
98 * - -ENOTSUP if this command is not supported.
99 * - Otherwise, error code from backend.
100 */
101 COREDUMP_CMD_ERASE_STORED_DUMP,
102
103 /**
104 * Copy the raw stored coredump.
105 *
106 * Returns:
107 * - copied size if successful
108 * - 0 if stored coredump is not found
109 * - -ENOTSUP if this command is not supported.
110 * - Otherwise, error code from backend.
111 */
112 COREDUMP_CMD_COPY_STORED_DUMP,
113
114 /**
115 * Invalidate the stored coredump. This is faster than
116 * erasing the whole partition.
117 *
118 * Returns:
119 * - 0 if successful.
120 * - -ENOTSUP if this command is not supported.
121 * - Otherwise, error code from backend.
122 */
123 COREDUMP_CMD_INVALIDATE_STORED_DUMP,
124
125 /**
126 * Max value for command ID.
127 */
128 COREDUMP_CMD_MAX
129 };
