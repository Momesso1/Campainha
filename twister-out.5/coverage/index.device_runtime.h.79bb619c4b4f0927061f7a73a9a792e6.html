32 /**
33 * @brief Automatically enable device runtime based on devicetree properties
34 *
35 * @note Must not be called from application code. See the
36 * zephyr,pm-device-runtime-auto property in pm.yaml and z_sys_init_run_level.
37 *
38 * @param dev Device instance.
39 *
40 * @return 0 if the device runtime PM is enabled successfully or it has not
41 * been requested for this device in devicetree, negative errno value on failure.
42 */
43 int pm_device_runtime_auto_enable(const struct device *dev);
44
45 /**
46 * @brief Enable device runtime PM
47 *
48 * This function will enable runtime PM on the given device. If the device is
49 * in #PM_DEVICE_STATE_ACTIVE state, the device will be suspended.
50 *
51 * @pre_kernel_ok
52 *
53 * @param dev Device instance.
54 *
55 * @return 0 on success, negative errno value on failure.
56 * @retval -EBUSY Device is busy.
57 * @retval -ENOTSUP Device does not support PM.
58 *
59 * @see pm_device_init_suspended()
60 */
61 int pm_device_runtime_enable(const struct device *dev);
62
63 /**
64 * @brief Disable device runtime PM
65 *
66 * If the device is currently suspended it will be resumed.
67 *
68 * @pre_kernel_ok
69 *
70 * @param dev Device instance.
71 *
72 * @return 0 on success, negative errno value on failure.
73 * @retval -ENOTSUP Device does not support PM.
74 */
75 int pm_device_runtime_disable(const struct device *dev);
76
77 /**
78 * @brief Resume a device based on usage count.
79 *
80 * This function will resume the device if the device is suspended (usage count
81 * equal to 0). In case of a resume failure, usage count and device state will
82 * be left unchanged. In all other cases, usage count will be incremented.
83 *
84 * If the device is still being suspended as a result of calling
85 * pm_device_runtime_put_async(), this function will wait for the operation to
86 * finish to then resume the device.
87 *
88 * @note It is safe to use this function in contexts where blocking is not
89 * allowed, e.g. ISR, provided the device PM implementation does not block.
90 *
91 * @pre_kernel_ok
92 *
93 * @param dev Device instance.
94 *
95 * @return 0 on success (including when runtime PM is not enabled or not
96 * available), negative errno value on failure.
97 * @retval -EWOULDBLOCK Call would block but it is not allowed (e.g. in ISR).
98 */
99 int pm_device_runtime_get(const struct device *dev);
100
101 /**
102 * @brief Suspend a device based on usage count.
103 *
104 * This function will suspend the device if the device is no longer required
105 * (usage count equal to 0). In case of suspend failure, usage count and device
106 * state will be left unchanged. In all other cases, usage count will be
107 * decremented (down to 0).
108 *
109 * @pre_kernel_ok
110 *
111 * @param dev Device instance.
112 *
113 * @return 0 on success (including when runtime PM is not enabled or not
114 * available), negative errno value on failure.
115 * @retval -EALREADY Device is already suspended (can only happen if get/put
116 * calls are unbalanced).
117 *
118 * @see pm_device_runtime_put_async()
119 */
120 int pm_device_runtime_put(const struct device *dev);
121
122 /**
123 * @brief Suspend a device based on usage count (asynchronously).
124 *
125 * This function will schedule the device suspension if the device is no longer
126 * required (usage count equal to 0). In all other cases, usage count will be
127 * decremented (down to 0).
128 *
129 * @note Asynchronous operations are not supported when in pre-kernel mode. In
130 * this case, the function will be blocking (equivalent to
131 * pm_device_runtime_put()).
132 *
133 * @pre_kernel_ok @async @isr_ok
134 *
135 * @param dev Device instance.
136 * @param delay Minimum amount of time before triggering the action.
137 *
138 * @return 0 on success (including when runtime PM is not enabled or not
139 * available), negative errno value on failure.
140 * @retval -EBUSY Device is busy.
141 * @retval -EALREADY Device is already suspended (can only happen if get/put
142 * calls are unbalanced).
143 *
144 * @see pm_device_runtime_put()
145 */
146 int pm_device_runtime_put_async(const struct device *dev, k_timeout_t delay);
147
148 /**
149 * @brief Check if device runtime is enabled for a given device.
150 *
151 * @pre_kernel_ok
152 *
153 * @param dev Device instance.
154 *
155 * @retval true Device has device runtime PM enabled.
156 * @retval false Device has device runtime PM disabled.
157 *
158 * @see pm_device_runtime_enable()
159 */
160 bool pm_device_runtime_is_enabled(const struct device *dev);
161
162 /**
163 * @brief Return the current device usage counter.
164 *
165 * @param dev Device instance.
166 *
167 * @return The current usage counter.
168 * @retval -ENOTSUP Device is not using runtime PM.
169 * @retval -ENOSYS Runtime PM is not enabled in this build

171 */
172 int pm_device_runtime_usage(const struct device *dev);
